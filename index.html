Backbone.js Dropdown View

Easily create dropdown (or popover) views for menus, inline editing, or just a simple tooltip. Has support for many locations (bottomLeft, bottom, bottomRight; on all sides)

How to use

Creating a Dropdown is very simple.


/*
	Simple Tooltip
*/
	$('#simple-tooltip').dropdown(
				'This is a tooltip', 
				{align: 'topRight', 
				trigger: 'hover'});
	
	
/*
	Menu
*/
	$('#menu').dropdown([
		{
			label:'Item 1', 
			onClick: function(){ 
				alert('item 1 clicked') 
			}
		},{
			label:'Item 2',
			onClick: function(){
				alert('item 2 clicked')
			}
		},{
			label:'Item 3', 
			onClick: function(){
				alert('item 3 clicked')
			}
		}
	], { w:70, align: 'bottom'})
	
	
/*
	Menu with Dividers
*/
	$('#menu2').dropdown([
		{'divider': 'Options'},
		{label:'Item 1', onClick: function(){ alert('item 1 clicked') }},
		{label:'Item 2', onClick: function(){ alert('item 2 clicked') }},
		'divider',
		{label:'Item 3', onClick: function(){ alert('item 3 clicked') }},
		{label:'Item 4', onClick: function(){ alert('item 4 clicked') }}
	], { w:70, align: 'bottom'})
	
	
/*
	Menu with event trigger
*/
	var menu3 = new Dropdown({
		renderTo: $('#menu3'),
		view: [
			{label:'Item 1', val: '1', onClick: 'trigger'},
			{label:'Item 2', val: '2', onClick: 'trigger'},
			{label:'Item 3', val: '3', onClick: 'trigger'}
		],
		w:70,
		align: 'bottom'
	})

	menu3.view.on('click', function(item, $li){
		alert('item '+item.val+' clicked')
	})
